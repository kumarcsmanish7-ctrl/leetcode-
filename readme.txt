# Go to your project folder
cd your-project-folder

# Initialize git (only first time)
git init

# Add your GitHub repository
git remote add origin https://github.com/kumarcsmanish7-ctrl/leetcode-.git

# Add files (all files)
git add .

# OR add specific file
# git add filename.java

# Commit your code
git commit -m "Added solutions"

# Set branch to main
git branch -M main

# Push to GitHub
git push -u origin main
